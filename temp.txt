CREATE TABLE "folder" (
	"id"	INTEGER NOT NULL,
	"name"	TEXT NOT NULL,
	PRIMARY KEY("id" AUTOINCREMENT)
)

CREATE TABLE "recipe" (
	"id"	INTEGER NOT NULL,
	"name"	TEXT NOT NULL,
	"instructions"	TEXT,
	"notes"	TEXT,
	"description"	TEXT,
	PRIMARY KEY("id" AUTOINCREMENT)
)

CREATE TABLE "folderRecipe" (
	"folderId"	INTEGER NOT NULL,
	"recipeId"	INTEGER NOT NULL,
	FOREIGN KEY("recipeId") REFERENCES "recipe"("id"),
	FOREIGN KEY("folderId") REFERENCES "folder"("id"),
	CONSTRAINT "PK_folder" PRIMARY KEY("recipeId","folderId")
)

CREATE TABLE "ingredientGroup" (
	"id"	INTEGER NOT NULL,
	"groupName"	TEXT NOT NULL,
	"recipeId"	INTEGER,
	FOREIGN KEY("recipeId") REFERENCES "recipe"("id") ON DELETE CASCADE,
	PRIMARY KEY("id" AUTOINCREMENT)
)

CREATE TABLE "ingredient" (
	"id"	INTEGER NOT NULL,
	"item"	TEXT NOT NULL,
	"measurement"	TEXT NOT NULL,
	"ingredientGroupId"	INTEGER,
	FOREIGN KEY("ingredientGroupId") REFERENCES "ingredientGroup"("id") ON DELETE CASCADE,
	PRIMARY KEY("id" AUTOINCREMENT)
)
